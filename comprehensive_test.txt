print("=== Comparison Operators ===")
print("\n")
(x == 42)?
print("x equals 42")
done

(y > 3)?
print("y is greater than 3")
done

(x != 0)?
print("x is not zero")
done

(y < 5)?
print("y is less than 5")
done

(x >= 40)?
print("x is greater than or equal to 40")
done

(y <= 4)?
print("y is less than or equal to 4")
done

print("=== Logical Operators ===")
print("\n")
(x > 30 && y < 5)?
print("AND: both conditions true")
done

(x < 30 || y > 2)?
print("OR: at least one condition true")
done

(x > 40 && y > 5)?
print("Should not print")
done

print("=== Conditional Blocks ===")
print("\n")
(x + y > 40)?
print("sum is greater than 40")
done

(z == 100)?
print("z equals 100")
done

print("=== Simple Loop ===")
print("\n")
let i: number = 1;
until (i <= 5)?
  print(i)
  print("\n")
  i += 1
done

print("=== Nested Loops ===")
print("\n")
let a: number = 1;
until (a <= 2)?
  print(a)
  print("\n")
  let b: number = 1;
  until (b <= 3)?
    print(b)
    print("\n")
    b += 1
  done
  a += 1
done

print("=== Loop with Compound Operators ===")
print("\n")
let counter2: number = 10;
until (counter2 >= 5)?
  print(counter2)
  print("\n")
  counter2 -= 2
done

print("=== Variable Scope in Nested Loops ===")
print("\n")
let outer: number = 1;
until (outer <= 2)?
  let inner: number = 100;
  until (inner <= 102)?
    print(inner)
    print("\n")
    inner += 1
  done
  outer += 1
done
